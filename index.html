1. 버블 정렬 (Bubble Sort)

#include <stdio.h>
int main() {
    int n, t; scanf("%d", &n);
    int arr[100];
    for(int i=0; i<n; i++) scanf("%d", &arr[i]);
    for(int i=0; i<n-1; i++) {
        for(int j=0; j<n-i-1; j++) {
            if(arr[j] > arr[j+1]) {
                t = arr[j]; arr[j] = arr[j+1]; arr[j+1] = t;
            }
        }
    }
    for(int i=0; i<n; i++) printf("%d ", arr[i]);
    return 0;
}
Input: 5 / 5 2 4 1 3 -> Output: 1 2 3 4 5

2. 최빈값 찾기 (Mode)

#include <stdio.h>
int main() {
    int n, maxCount = 0, mode = 0; scanf("%d", &n);
    int count[101] = {0};
    for(int i=0; i<n; i++) {
        int val; scanf("%d", &val);
        count[val]++;
        if(count[val] > maxCount) { maxCount = count[val]; mode = val; }
    }
    printf("최빈값: %d", mode);
    return 0;
}
Input: 5 / 1 2 2 3 1 -> Output: 최빈값: 1

#include <stdio.h>
void swap(int *a, int *b) {
    int temp = *a; *a = *b; *b = temp;
}
int main() {
    int x = 10, y = 20;
    swap(&x, &y);
    printf("Swap: %d %d\n", x, y);
    int arr[3] = {1, 2, 3};
    int *p = arr;
    for(int i=0; i<3; i++) printf("%d ", *(p+i));
    return 0;
}

4. 행렬 곱셈 (Matrix Multiplication)

#include <stdio.h>
int main() {
    int r1, c1, r2, c2; scanf("%d %d %d %d", &r1, &c1, &r2, &c2);
    int a[10][10], b[10][10], res[10][10] = {0};
    for(int i=0; i<r1; i++) for(int j=0; j<c1; j++) scanf("%d", &a[i][j]);
    for(int i=0; i<r2; i++) for(int j=0; j<c2; j++) scanf("%d", &b[i][j]);
    for(int i=0; i<r1; i++) for(int j=0; j<c2; j++) for(int k=0; k<c1; k++) res[i][j] += a[i][k] * b[k][j];
    for(int i=0; i<r1; i++) { for(int j=0; j<c2; j++) printf("%d ", res[i][j]); printf("\n"); }
    return 0;
}

5. 두 수 사이의 소수 (제곱근 최적화)

#include <stdio.h>
int main() {
    int start, end; scanf("%d %d", &start, &end);
    for (int i = start; i <= end; i++) {
        if (i < 2) continue;
        int is_prime = 1;
        for (int j = 2; j * j <= i; j++) { // 제곱근까지만 확인하여 성능 향상
            if (i % j == 0) { is_prime = 0; break; }
        }
        if (is_prime) printf("%d ", i);
    }
    return 0;
}
Input: 10 30 -> Output: 11 13 17 19 23 29

1. 선택 정렬 (Selection Sort)

#include <iostream>
#include <vector>
using namespace std;
int main() {
    int n; cin >> n; vector<int> v(n);
    for(int i=0; i<n; i++) cin >> v[i];
    for(int i=0; i<n-1; i++) {
        int m = i;
        for(int j=i+1; j<n; j++) if(v[j] < v[m]) m = j;
        swap(v[i], v[m]);
    }
    for(int x : v) cout << x << " ";
    return 0;
}
Input: 4 / 4 1 3 2 -> Output: 1 2 3 4

5. 두 수 사이의 소수 (에라토스테네스의 체)

#include <iostream>
#include <vector>
using namespace std;
int main() {
    int s, e; cin >> s >> e;
    vector<bool> is_prime(e + 1, true);
    is_prime[0] = is_prime[1] = false;
    for (int i = 2; i * i <= e; i++) {
        if (is_prime[i]) {
            for (int j = i * i; j <= e; j += i) is_prime[j] = false;
        }
    }
    for (int i = s; i <= e; i++) if (is_prime[i]) cout << i << " ";
    return 0;
}
Input: 1 20 -> Output: 2 3 5 7 11 13 17 19
