C

1. 버블 정렬 및 최빈값

#include <stdio.h>
int main() {
    int n, maxCount = 0, mode = 0;
    scanf("%d", &n);
    int arr[n], count[101] = {0};
    for(int i=0; i<n; i++) {
        scanf("%d", &arr[i]);
        count[arr[i]]++;
        if(count[arr[i]] > maxCount) { maxCount = count[arr[i]]; mode = arr[i]; }
    }
    printf("최빈값: %d", mode);
    return 0;
}
Input: 5 / 1 2 2 3 4
Output: 최빈값: 2

2. 포인터 값 교환 (Swap)

void swap(int *a, int *b) {
    int temp = *a; *a = *b; *b = temp;
}
int main() {
    int x, y; scanf("%d %d", &x, &y);
    swap(&x, &y); printf("%d %d", x, y);
    return 0;
}

1. 선택 정렬 및 최빈값

#include <iostream>
#include <map>
using namespace std;
int main() {
    int n, x, maxC = 0, mode; cin >> n;
    map<int, int> m;
    for(int i=0; i<n; i++) { cin >> x; m[x]++; if(m[x] > maxC) { maxC = m[x]; mode = x; } }
    cout << "최빈값: " << mode;
    return 0;
}

int main() {
    int n; cin >> n;
    int arr[100]; int *ptr = arr;
    for(int i=0; i<n; i++) cin >> *(ptr + i);
