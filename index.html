버블 정렬 (Bubble Sort)

#include <stdio.h>

int main() {
    int n, temp;
    scanf("%d", &n);
    int arr[n];
    for(int i=0; i<n; i++) scanf("%d", &arr[i]);

    for(int i=0; i<n-1; i++) {
        for(int j=0; j<n-i-1; j++) {
            if(arr[j] > arr[j+1]) {
                temp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = temp;
            }
        }
    }
    for(int i=0; i<n; i++) printf("%d ", arr[i]);
    return 0;
}
입출력 예제
Input: 5 / 5 2 4 1 3
Output: 1 2 3 4 5

포인터 값 교환 (Swap)

#include <stdio.h>

void swap(int *a, int *b) {
    int temp = *a;
    *a = *b;
    *b = temp;
}

int main() {
    int x, y;
    scanf("%d %d", &x, &y);
    swap(&x, &y);
    printf("%d %d", x, y);
    return 0;
}
C++

선택 정렬 (Selection Sort)

#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n; cin >> n;
    vector<int> a(n);
    for(int i=0; i<n; i++) cin >> a[i];

    for(int i=0; i<n-1; i++) {
        int min_idx = i;
        for(int j=i+1; j<n; j++)
            if(a[j] < a[min_idx]) min_idx = j;
        swap(a[i], a[min_idx]);
    }
    for(int x : a) cout << x << " ";
    return 0;
}

#include <iostream>
using namespace std;

int main() {
    int n; cin >> n;
    int arr[100];
    int *ptr = arr; 

    for(int i=0; i<n; i++) cin >> *(ptr + i);

    for(int i=n-1; i>=0; i--) cout << *(ptr + i) << " ";
    return 0;
}
C#

배열 입력 및 평균 계산

using System;
using System.Linq;

class Program {
    static void Main() {
        int n = int.Parse(Console.ReadLine());
        int[] numbers = Console.ReadLine().Split(' ')
                               .Select(int.Parse).ToArray();

        double average = numbers.Average();
        Console.WriteLine($"{n}개의 평균: {average:F2}");
    }
}
