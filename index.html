버블 정렬 (Bubble Sort)

Source Code
#include <stdio.h>

int main() {
    int n, temp;
    scanf("%d", &n);
    int arr[n];
    for(int i=0; i<n; i++) scanf("%d", &arr[i]);

    for(int i=0; i<n-1; i++) {
        for(int j=0; j<n-i-1; j++) {
            if(arr[j] > arr[j+1]) {
                temp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = temp;
            }
        }
    }
    for(int i=0; i<n; i++) printf("%d ", arr[i]);
    return 0;
}
Input Example
5 5 2 4 1 3
Output Example
1 2 3 4 5

선택 정렬 (Selection Sort)

Source Code
#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<int> a(n);
    for(int i=0; i<n; i++) cin >> a[i];

    for(int i=0; i<n-1; i++) {
        int min_idx = i;
        for(int j=i+1; j<n; j++)
            if(a[j] < a[min_idx]) min_idx = j;
        swap(a[i], a[min_idx]);
    }

    for(int i=0; i<n; i++) cout << a[i] << " ";
    return 0;
}
Input Example
6 10 5 8 2 1 7
Output Example
1 2 5 7 8 10
