버블 정렬 (Bubble Sort)

#include <stdio.h>
int main() {
    int n; scanf("%d", &n);
    int arr[n];
    for(int i=0; i<n; i++) scanf("%d", &arr[i]);
    for(int i=0; i<n-1; i++) {
        for(int j=0; j<n-i-1; j++) {
            if(arr[j] > arr[j+1]) {
                int t = arr[j]; arr[j] = arr[j+1]; arr[j+1] = t;
            }
        }
    }
    for(int i=0; i<n; i++) printf("%d ", arr[i]);
    return 0;
}
Input: 3 / 30 10 20
Output: 10 20 30

포인터 값 교환 (Swap)

#include <stdio.h>
void swap(int *a, int *b) {
    int temp = *a; *a = *b; *b = temp;
}
int main() {
    int x, y; scanf("%d %d", &x, &y);
    swap(&x, &y);
    printf("%d %d", x, y);
    return 0;
}
Input: 5 9
Output: 9 5
C++

선택 정렬 (Selection Sort)

#include <iostream>
#include <vector>
using namespace std;
int main() {
    int n; cin >> n;
    vector<int> v(n);
    for(int i=0; i<n; i++) cin >> v[i];
    for(int i=0; i<n-1; i++) {
        int m = i;
        for(int j=i+1; j<n; j++) if(v[j] < v[m]) m = j;
        swap(v[i], v[m]);
    }
    for(int x : v) cout << x << " ";
    return 0;
}
Input: 4 / 4 1 3 2
Output: 1 2 3 4
