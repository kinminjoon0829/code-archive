1. 버블 정렬 (Bubble Sort)

#include <stdio.h>
int main() {
    int n; scanf("%d", &n);
    int arr[n];
    for(int i=0; i<n; i++) scanf("%d", &arr[i]);
    for(int i=0; i<n-1; i++) {
        for(int j=0; j<n-i-1; j++) {
            if(arr[j] > arr[j+1]) {
                int t = arr[j]; arr[j] = arr[j+1]; arr[j+1] = t;
            }
        }
    }
    for(int i=0; i<n; i++) printf("%d ", arr[i]);
    return 0;
}
Input: 3 / 30 10 20
Output: 10 20 30

2. 최빈값 구하기 (Mode)

#include <stdio.h>
int main() {
    int n, maxCount = 0, mode = 0;
    scanf("%d", &n);
    int arr[n], count[101] = {0}; // 0~100 범위 가정
    for(int i=0; i<n; i++) {
        scanf("%d", &arr[i]);
        count[arr[i]]++;
        if(count[arr[i]] > maxCount) {
            maxCount = count[arr[i]];
            mode = arr[i];
        }
    }
    printf("최빈값: %d", mode);
    return 0;
}

1. 선택 정렬 (Selection Sort)

#include <iostream>
#include <vector>
using namespace std;
int main() {
    int n; cin >> n;
    vector<int> v(n);
    for(int i=0; i<n; i++) cin >> v[i];
    for(int i=0; i<n-1; i++) {
        int m = i;
        for(int j=i+1; j<n; j++) if(v[j] < v[m]) m = j;
        swap(v[i], v[m]);
    }
    for(int x : v) cout << x << " ";
    return 0;
}

#include <iostream>

int main() {
    int n; cin >> n;
    int arr[100]; int *ptr = arr;
    for(int i=0; i<n; i++) cin >> *(ptr + i);

3. 최빈값 구하기 (Mode)

#include <iostream>
#include <map>
using namespace std;
int main() {
    int n, x, maxC = 0, mode; cin >> n;
    map<int, int> m;
    for(int i=0; i<n; i++) {
        cin >> x; m[x]++;
        if(m[x] > maxC) { maxC = m[x]; mode = x; }
    }
    cout << "최빈값: " << mode;
    return 0;
}
