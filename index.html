C

1. 버블 정렬 (Bubble Sort)

#include <stdio.h>
int main() {
    int n; scanf("%d", &n);
    int arr[n];
    for(int i=0; i<n; i++) scanf("%d", &arr[i]);
    for(int i=0; i<n-1; i++) {
        for(int j=0; j<n-i-1; j++) {
            if(arr[j] > arr[j+1]) {
                int t = arr[j]; arr[j] = arr[j+1]; arr[j+1] = t;
            }
        }
    }
    for(int i=0; i<n; i++) printf("%d ", arr[i]);
    return 0;
}
Input: 5 / 5 1 4 2 3 -> Output: 1 2 3 4 5

2. 최빈값 찾기 (Mode)

#include <stdio.h>
int main() {
    int n, maxCount = 0, mode = 0; scanf("%d", &n);
    int count[101] = {0}; 
    for(int i=0; i<n; i++) {
        int val; scanf("%d", &val);
        count[val]++;
        if(count[val] > maxCount) { maxCount = count[val]; mode = val; }
    }
    printf("최빈값: %d (빈도: %d)", mode, maxCount);
    return 0;
}

3. 두 수 사이의 소수 출력

#include <stdio.h>
int main() {
    int a, b; scanf("%d %d", &a, &b);
    for(int i=a; i<=b; i++) {
        int isPrime = (i > 1);
        for(int j=2; j*j<=i; j++) {
            if(i % j == 0) { isPrime = 0; break; }
        }
        if(isPrime) printf("%d ", i);
    }
    return 0;
}
Input: 10 20 -> Output: 11 13 17 19
C++

1. 선택 정렬 (Selection Sort)

#include <iostream>
#include <vector>
using namespace std;
int main() {
    int n; cin >> n; vector<int> v(n);
    for(int i=0; i<n; i++) cin >> v[i];
    for(int i=0; i<n-1; i++) {
        int m = i;
        for(int j=i+1; j<n; j++) if(v[j] < v[m]) m = j;
        swap(v[i], v[m]);
    }
    for(int x : v) cout << x << " ";
    return 0;
}

2. 최빈값 찾기 (Map 활용)

#include <iostream>
#include <map>
using namespace std;
int main() {
    int n, x, maxC = 0, mode; cin >> n;
    map<int, int> m;
    for(int i=0; i<n; i++) { cin >> x; m[x]++; if(m[x] > maxC) { maxC = m[x]; mode = x; } }
    cout << "최빈값: " << mode;
    return 0;
}

3. 두 수 사이의 소수 출력

#include <iostream>
using namespace std;
bool check(int n) {
    if(n < 2) return false;
    for(int i=2; i*i<=n; i++) if(n % i == 0) return false;
    return true;
}
int main() {
    int s, e; cin >> s >> e;
    for(int i=s; i<=e; i++) if(check(i)) cout << i << " ";
    return 0;
}
Input: 1 10 -> Output: 2 3 5 7

C#

배열 평균 계산 및 입출력

using System;
using System.Linq;

    static void Main() {

2. 두 수 사이의 소수 (List Comprehension)

a, b = map(int, input().split())

print(*primes)
