포인터 활용 값 교환 (Swap)

Source Code
#include <stdio.h>

void swap(int *a, int *b) {
    int temp = *a;
    *a = *b;
    *b = temp;
}

int main() {
    int x, y;
    scanf("%d %d", &x, &y);
    swap(&x, &y);
    printf("%d %d", x, y);
    return 0;
}
Input Example
10 20
Output Example
20 10

포인터로 배열 접근 및 역순 출력

Source Code
#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;
    int arr[100];
    int *ptr = arr; 

    for(int i=0; i<n; i++) cin >> *(ptr + i);

    for(int i=n-1; i>=0; i--) cout << *(ptr + i) << " ";
    return 0;
}
Input Example
3 1 2 3
Output Example
3 2 1
